# --- 1. НАЛАШТУВАННЯ ТА БЕЗПЕКА (Один раз на новому ПК) ---
git config --global user.name "Zakharko Adrian"   # Встановити ім'я
git config --global user.email "zakharkoadr@gmail.com" # Встановити пошту
git config --global user.name                     # Перевірити встановлене ім'я
git config --global user.email                    # Перевірити встановлену пошту

ssh-keygen -t rsa -C "zakharkoadr@gmail.com"      # Згенерувати SSH-ключ для GitHub
cat ~/.ssh/id_rsa.pub | clip                      # Скопіювати ключ у буфер обміну

# --- 2. СТАРТ ПРОЄКТУ (Один раз для папки) ---
git init                                          # Зробити папку репозиторієм
git remote add origin <LINK>                      # Прив'язати до GitHub (лінк репозиторію)
git clone <LINK>                                  # АБО: Скачати готовий проєкт з GitHub

# --- 3. ЩОДЕННИЙ РИТМ (База 90% часу) ---
git status                                        # Перевірити, що змінилося (червоне/зелене)
git add .                                         # Додати ВСІ зміни в "коробку" (індекс)
git commit -m "Опис змін"                         # Зберегти зміни (запакувати коробку)
git push                                          # Відправити зміни в інтернет (на GitHub)

# --- 4. ЯКЩО ПРАЦЮЄШ НЕ САМ ---
git pull                                          # Затягнути оновлення з GitHub собі

# --- 5. ГІЛКИ ТА ЗЛИТТЯ (BRANCHING & MERGING) ---
# Гілки дозволяють безпечно писати новий код, не ламаючи головний (master/main)

git branch                        # Показати список усіх локальних гілок (зірочка * покаже, де ти зараз)
git branch <назва_гілки>          # Створити нову гілку, але ЗАЛИШИТИСЯ на поточній
git checkout -b <назва_гілки>     # Створити нову гілку і ОДРАЗУ ПЕРЕЙТИ на неї (найчастіший варіант)
git checkout <назва_гілки>        # Перемкнутися на існуючу гілку (наприклад, git checkout master)

# ЗЛИТТЯ (MERGE) - об'єднання паралельних всесвітів
# Важливо: перед злиттям треба перейти в ту гілку, В ЯКУ ти хочеш затягнути код (зазвичай у master)
git merge <назва_гілки>           # Злити вказану гілку в ту, де ти зараз знаходишся (автоматично обере Fast-forward, якщо можливо)
git merge --no-ff <назва_гілки>   # ПРИМУСОВО створити Merge Commit (зберігає історію фічі, навіть якщо можна було зробити Fast-forward)
git merge --abort                 # ПАНІКА ПРИ ЗЛИТТІ: скасувати процес злиття, якщо виникли страшні конфлікти коду

# ПРИБИРАННЯ (коли фіча вже злита)
git branch -d <назва_гілки>       # Видалити гілку (безпечно: Гіт не дасть видалити, якщо код ще не злитий у master)
git branch -D <назва_гілки>       # ПРИМУСОВО видалити гілку (кнопка знищення незлитого коду)

# --- 6. КНОПКИ ПАНІКИ ТА ШВИДКА ДОПОМОГА (Обережно!) ---
git log                                           # Історія змін (натисни q щоб вийти)
git checkout .                                    # СКАСУВАТИ всі зміни у всіх файлах (які ще не в add)
git checkout -- <файл>                            # Скасувати незбережені зміни лише в конкретному файлі
git clean -xdf                                    # ЖОРСТКА ЧИСТКА: видалити всі нові файли, про які Git ще не знає
git reset -- <файл>                               # Прибрати файл зі "сцени" (індексу), якщо випадково зробив git add
git commit --amend -m "Новий текст"               # Переписати повідомлення останнього коміту / докинути файл
git revert <hash> - перші 7 символів              # Безпечно скасувати старий коміт (створює новий "анти-коміт")

# ⚠️ ТРИ РІВНІ СКИДАННЯ (RESET) ПOПЕРЕДНЬОГО КОМІТУ:
git reset HEAD^^             (^ - один коміт)     # Відмотати час на 2 коміти назад (стандартний mixed відкат)
git reset --soft HEAD~1                           # М'ЯКО: Скасувати коміт, але залишити файли підготовленими (в індексі)
git reset --mixed HEAD~1    (1 - один коміт)      # СЕРЕДНЬО: Скасувати коміт і прибрати з індексу (код у файлах залишається!)
git reset --hard HEAD~1                           # 🚨 ПОВНЕ ЗНИЩЕННЯ: Скасувати коміт і СТЕРТИ всі зміни у файлах

# --- 7. "ПІД КАПОТОМ" ГІТА (Анатомія бази даних) ---
git show -s --pretty=raw <hash>                   # Зазирнути всередину коміту (автор, дата, посилання на tree)
git ls-tree <hash>                                # Відкрити дерево (tree) і подивитися список файлів у ньому
git show <hash>                                   # Прочитати чистий текст конкретного файлу (blob)


# --- 8. ФЕЙСКОНТРОЛЬ (.gitignore) ---
# Файл .gitignore вказує Гіту, які файли НЕ треба відстежувати (кеш, збірки, паролі)
*.log                                             # Ігнорувати всі файли з розширенням .log
!error.log                                        # ВИНЯТОК: відстежувати error.log (навіть якщо *.log ігнорується)
/TODO                                             # Ігнорувати файл TODO тільки в кореневій папці
build/                                            # Ігнорувати повністю папку build/ та весь її вміст
doc/**/*.pdf                                      # Ігнорувати всі .pdf файли в папці doc/ та у всіх її підпапках


# --- 9. КОНФЛІКТИ ПРИ ЗЛИТТІ (MERGE CONFLICTS) ---
# Виникають, коли двоє людей змінили один і той самий рядок коду
git merge --abort                 # ПАНІКА: скасувати злиття і повернути все як було (до команди merge)
git diff                          # Подивитися, де саме виник конфлікт
git checkout --ours <файл>        # Швидке вирішення: залишити НАШУ версію коду (ту, де ми зараз)
git checkout --theirs <файл>      # Швидке вирішення: залишити ЇХНЮ версію (яку ми вливаємо)

# ПАМ'ЯТКА: Як уникати конфліктів у команді:
# 1. Роби маленькі і часті коміти.
# 2. Не міняй відступи/пробіли у файлах без потреби.
# 3. Регулярно підтягуй чужі зміни з master у свою гілку.


# --- 10. REBASE (ПЕРЕЗАПИС ІСТОРІЇ) ---
# Альтернатива merge. Робить історію ідеально прямою лінією.
git rebase master                 # "Відірвати" поточну гілку і причепити її на верхівку master
git rebase --abort                # ПАНІКА: скасувати rebase (якщо конфлікти зводять з розуму)
git rebase --continue             # Продовжити процес після того, як ти руками вирішив конфлікти у файлах

# 🚨 ЗОЛОТЕ ПРАВИЛО REBASE:
# НІКОЛИ не роби rebase публічних гілок (main/master), які вже є на сервері!
# Використовуй тільки для своїх локальних гілок перед тим, як вливати їх у master.



# --- 11. CHERRY-PICK (ЮВЕЛІРНА РОБОТА) ---
# Коли треба взяти лише один конкретний коміт з іншої гілки, не зливаючи гілки повністю.
git cherry-pick <hash>            # Скопіювати коміт із вказаним хешем у твою поточну гілку
git cherry-pick --abort           # ПАНІКА: скасувати операцію, якщо при копіюванні вилізли конфлікти
git cherry-pick --continue        # Продовжити після того, як ти руками вирішив конфлікти



# --- 12. ТЕГИ / ЗАКЛАДКИ (TAGS) ---
# Використовуються для фіксації готових версій проєкту (релізів, наприклад v1.0, v2.1)
git tag <назва_тегу>              # Поставити тег (закладку) на поточний коміт (наприклад, git tag v1.0)
git tag                           # Показати список усіх існуючих тегів (можна з прапорцем --list)
git push --tags                   # ВАЖЛИВО: звичайний push не відправляє теги на сервер (GitHub). Потрібна ця команда!
git checkout <назва_тегу>         # "Машина часу": повернутися і подивитися, як виглядав код у момент цього тегу




# --- 13. СХОВАНКА / ЗАНАЧКА (STASH) ---
# Рятує, коли треба терміново змінити гілку, але поточний код недописаний і комітити не можна.
git stash                         # Швидко сховати всі незбережені зміни в "кишеню"
git stash save "мій текст"        # Сховати з підписом (щоб не забути, що саме ти там залишив)
git stash list                    # Подивитися список всього, що лежить у твоїх схованках
git stash pop                     # Дістати останні зміни зі схованки НА СТІЛ (і видалити зі схованки)
git stash apply                   # Дістати зміни на стіл, але ЗАЛИШИТИ їх копію в схованці
git stash drop                    # Просто викинути ці збережені зміни в смітник



# --- 14. ВІДДАЛЕНІ РЕПОЗИТОРІЇ (REMOTES) ---
# Зв'язок твого локального коду з серверами (GitHub, GitLab тощо). 
# "origin" — це стандартна назва для твого головного сервера.

git remote -v                     # Показати список усіх підключених серверів та їхні URL-адреси
git remote add <назва> <url>      # Підключити локальний проєкт до сервера (напр., git remote add origin https://github.com/...)
git remote show <назва>           # Показати детальну інформацію про конкретний сервер (гілки, статуси)

# ОБМІН ДАНИМИ З СЕРВЕРОМ:
# git push   — Відправити свої локальні коміти на віддалений сервер
# git fetch  — Тихо завантажити нові коміти з сервера (щоб подивитися), БЕЗ злиття з твоїми файлами
# git pull   — Завантажити і ОДРАЗУ злити зміни з сервера у твою поточну гілку (fetch + merge)



# ==========================================
# --- 15. ПРОСУНУТІ ІНСТРУМЕНТИ (EXTRAS) ---
# ==========================================
# Команди Senior-рівня для розслідувань, кастомізації консолі та хардкору.

# 🕵️‍♂️ ДЕТЕКТИВ (Пошук винних і багів)
git blame <файл>                  # Показати автора, хеш і дату для КОЖНОГО рядка у файлі ("хто написав цей код?")
git bisect start                  # Запустити автоматичний бінарний пошук коміту-вбивці (який приніс баг)
git bisect bad                    # Сказати Гіту: "Зараз код зламаний"
git bisect good <hash>            # Сказати Гіту: "А от на цьому старому коміті все ще працювало" (Git сам знайде винного)

# 🎨 КРАСИВІ ЛОГИ ТА АНАЛІТИКА
git log --pretty=oneline          # Показати історію компактно (один коміт = один рядок, без зайвої води)
git log --oneline --graph --all   # МАСТХЕВ: Намалювати красиве дерево гілок і злиттів прямо в консолі
git log master..feature           # Показати ЩО НОВОГО: коміти, які є у гілці feature, але яких ще немає у master

# 🛠 КАСТОМІЗАЦІЯ (Налаштовуємо термінал під себе)
git config --global core.editor "code --wait"  # Зробити VS Code редактором для Гіта (щоб не страждати у Vim)
git config --global alias.last 'log -1 HEAD'   # Створити своє скорочення (тепер команда `git last` покаже останній коміт)
git config --global alias.st status            # Тепер замість довгого `git status` можна писати просто `git st`

# ☢️ ХАРДКОР ТА НЕБЕЗПЕЧНА ЗОНА
git filter-branch --tree-filter 'rm -f pass.txt' HEAD # ПОВНЕ ЗНИЩЕННЯ: Видалити файл з паролями з УСІЄЇ історії (змінить хеші всіх комітів!)
git rerere                        # "Розумний Git": запам'ятати, як ти вирішив конфлікт, щоб наступного разу він вирішив його сам
git submodule                     # Вставити інший (чужий) Git-репозиторій всередину твого проєкту як звичайну папку

1. Centralized Strategy (Централізована)
Як працює: Усі розробники пушать свій код безпосередньо в одну спільну гілку (master або main). Інші гілки просто не створюються.

Плюси: Максимально просто. Підходить для соло-розробки або мікро-проєктів на 1-2 людини.

Мінуси: "Пекло конфліктів". Якщо хтось залив код з помилкою, master ламається у всіх. Працювати паралельно над одним файлом практично неможливо.

Вердикт DevOps: Застарілий підхід з епохи SVN. У сучасному комерційному ІТ майже не використовується.

2. Feature-branch Workflow (Гілкування під фічі)
Як працює: master — це стабільний робочий код. Для кожної нової задачі (кнопка, багфікс, сторінка) розробник створює окрему тимчасову гілку від master (наприклад, feature/login). Після завершення роботи код вливається назад у master через Pull Request.

Плюси: Ідеальна пісочниця. Твій неробочий код нікому не заважає. Зручно проводити Code Review (перевірку коду колегами) перед злиттям.

Мінуси: Якщо гілка живе занадто довго (тижнями), її буде дуже боляче зливати з master через накопичені відмінності.

Вердикт DevOps: Золотий стандарт сучасної розробки. Використовується у 90% компаній для веб-проєктів.

3. Gitflow (Корпоративний монстр)
Як працює: Жорстка структура з чіткими ролями для гілок.

master: тільки продакшен (те, що бачить клієнт). Кожен коміт має тег версії (v1.0, v2.1).

develop: головна гілка для розробників. Сюди зливаються всі готові фічі.

feature/...: гілки для задач, які відгалужуються від develop (не від master!).

hotfix/...: пожежні гілки для критичних багів на проді. Відгалужуються від master, виправляються і вливаються одночасно в master та develop.

Плюси: Максимальний контроль над релізами. Ідеально для банків, ентерпрайз-систем та ігор.

Мінуси: Занадто складно і багато бюрократії (зайвих дій у терміналі) для простих проєктів.

Вердикт DevOps: Потужний інструмент для великих команд зі строгими графіками випуску оновлень.


4. Integration Manager Workflow (Менеджер інтеграції)
Як працює: Є головний репозиторій ("blessed repository"), куди записується офіційна історія. Але ніхто з розробників не має туди прямого доступу на push. Кожен розробник має свій власний публічний репозиторій. Ти пишеш код, пушиш у свій репозиторій, а потім кажеш Менеджеру (головному розробнику): "Ей, я там зробив круту фічу, забери її собі". Менеджер стягує твій код, тестує його і, якщо все ок, пушить у "blessed repository".

Коли використовувати: Це класична схема для ओपन-сорс (Open Source) проєктів до епохи сучасного GitHub.

Вердикт DevOps: Дуже безпечно. Головний репозиторій ніколи не зламається через випадковий пуш новачка, бо є "фейсконтроль".

5. Dictator and Lieutenants Workflow (Диктатор і Лейтенанти)
Як працює: Це масштабована версія попереднього підходу для проєктів-гігантів (наприклад, розробка ядра Linux, де працюють тисячі людей). Уяви собі класичну феодальну структуру з Вестеросу: є звичайні лорди (розробники), які пишуть локальний код. Вони передають свої напрацювання Хранителям Півночі чи Сходу (Лейтенантам), які відповідають за цілі великі підсистеми (наприклад, мережеві драйвери). А вже ці Хранителі збирають найкращий код і несуть його безпосередньо до Короля на Залізному Троні (Диктатора), який єдиний має право вливати зміни у головний літопис (blessed repository).

Коли використовувати: Тільки для величезних проєктів, де один менеджер просто фізично не встигне перевірити код від 10 000 розробників.

Вердикт DevOps: Крута ієрархія, але для 99.9% комерційних компаній це оверкіл.

6. Forking Workflow (Форк-модель)
Як працює: Це сучасна, автоматизована версія Integration Manager, створена спеціально для GitHub/GitLab. Замість того, щоб вручну налаштовувати публічні репозиторії, ти просто натискаєш кнопку "Fork" на сайті. GitHub робить повну копію чужого проєкту у твій акаунт. Ти працюєш у цій копії, створюєш коміти, а коли готово — натискаєш кнопку "Pull Request". Власник оригіналу бачить твій запит і натискає "Merge", вливаючи твій код у свій оригінальний репозиторій.

Коли використовувати: Будь-який Open Source на GitHub. Якщо ти захочеш виправити баг у бібліотеці React чи Angular — ти будеш робити це саме через Fork.

Вердикт DevOps: Це те, що ти обов'язково маєш вміти. Навіть на деяких закритих корпоративних проєктах використовують форки для додаткової безпеки.
